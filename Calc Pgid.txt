Private Sub ComboBox2_Change()

ComboBox3.AddItem "Сталь 10"

End If

End If

End Sub

Private Sub CommandButton1_Click()
Dim D As Double
Dim S As Double
Dim R As Double
Dim P As Double
Dim NPD As Double
Dim Steel As Double
Dim ERROR As Boolean

ERROR = False

If ComboBox1.Value = "выберите из списка" Then
D = MsgBox("Ошибочка, выберите Диаметр трубы!", vbOKOnly, "Ошибка")
ERROR = True
Else
D = ComboBox1.Value
Label8.Caption = D
End If


If ComboBox2.Value = "выберите из списка" Then
NPD = MsgBox("Ошибочка, выберите тип нормативной документации!", vbOKOnly, "Ошибка")
ERROR = True
Else
    If ComboBox2.Value = "ГОСТ" Then
    NPD = 0.4 * 1
    Label7.Caption = NPD
    End If
    If ComboBox2.Value = "ТУ" Then
    NPD = 0.8 * 1
    Label7.Caption = NPD
    End If
End If

If ComboBox3.Value = "выберите из списка" Or ComboBox3.Value = "" Then
NPD = MsgBox("Ошибочка, выберите сталь!", vbOKOnly, "Ошибка")
ERROR = True
Else
    If ComboBox3.Value = "Сталь 10" Then
    Steel = 36
    Label10.Caption = Steel
    End If
    If ComboBox3.Value = "Сталь 20" Then
    Steel = 42
    Label10.Caption = Steel
    End If
    If ComboBox3.Value = "Сталь 09Г2С" Then
    Steel = 48
    Label10.Caption = Steel
    End If
    If ComboBox3.Value = "Сталь 20А" Then
    Steel = 34.5
    Label10.Caption = Steel
    End If
    If ComboBox3.Value = "Сталь 13ХФА" Then
    Steel = 38
    Label10.Caption = Steel
    End If
End If

If IsNumeric(TextBox2.Value) = False Then
S = MsgBox("Ошибочка, введите толщину стенки в мм, число. Для разделения целой и дробной частей исп запятую", vbOKOnly, "Ошибка")
ERROR = True
Else
S = TextBox2.Value
Label9.Caption = S
End If

If ERROR = False Then

    If TextBox2.Text / ComboBox1.Value > 0.13 Then
    S = MsgBox("Ошибочка, P исп для вашей трубы должно считаться по другой формуле!", vbOKOnly, "Ошибка")
    ERROR = True
    End If
    
End If

If ERROR = False Then
P = Round(200 * S * 0.875 * NPD * Steel / (D - S), 2)
TextBox4.Text = P
TextBox5.Text = Round(P * 0.0980665, 2)
End If


End Sub
